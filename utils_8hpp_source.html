549 void readMNIST(const string filename,
550  MyMatrix &images,
551  bool invert_pixel){
552 
553  ifstream file(filename.c_str(), ios::binary);
554  if (file.is_open()){
555  unsigned n_training = 0;
556  int magic_number = 0;
557  int n_rows = 0;
558  int n_cols = 0;
559 
560  file.read((char*) &magic_number, sizeof(magic_number));
561  magic_number = ReverseInt(magic_number);
562  file.read((char*) &n_training,sizeof(n_training));
563  n_training = ReverseInt(n_training);
564  file.read((char*) &n_rows, sizeof(n_rows));
565  n_rows = ReverseInt(n_rows);
566  file.read((char*) &n_cols, sizeof(n_cols));
567  n_cols = ReverseInt(n_cols);
568 
569  images.resize(n_training, n_rows*n_cols);
570 
571  for(int i = 0; i < n_training; ++i){
572  for(int r = 0; r < n_rows; ++r){
573  for(int c = 0; c < n_cols; ++c){
574  unsigned char temp = 0;
575  file.read((char*) &temp, sizeof(temp));
576  if(!invert_pixel)
577  images(i,r*n_cols+c) = (double) temp /255.;
578  else{
579  images(i,r*n_cols+c) = (1.0 - (double) temp /255.);
580  }
581  }
582  }
583  }
584  }
