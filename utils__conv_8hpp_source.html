169 void buildConvMatrix(const unsigned n_img,
170  const unsigned conv_N1,
171  const unsigned conv_N2,
172  const unsigned F,
173  const unsigned S,
174  const unsigned P,
175  const MyMatrix& conv_layer,
176  MyMatrix& conv_matrix){
177 
178  const unsigned conv_depth = conv_layer.rows();
179  conv_matrix.setZero(conv_depth*F*F,n_img*conv_N2*conv_N2);
180  for(unsigned i = 0; i < n_img; i++){
181  for(unsigned j = 0; j < conv_depth; j++){
182  for(unsigned no = 0; no < conv_N2; no++){
183  for(unsigned ni = 0; ni < conv_N2; ni++){
184  const unsigned ko_start = std::max(0,(int)P-(int)no);
185  const unsigned ko_end = F-std::max(0,(int)P-(int)conv_N2+(int)no+1);
186  const unsigned ki_start = std::max(0,(int)P-(int)ni);
187  const unsigned ki_end = F-std::max(0,(int)P-(int)conv_N2+(int)ni+1);
188  for(unsigned ko = ko_start; ko < ko_end; ko++){
189  for(unsigned ki = ki_start; ki < ki_end; ki++){
190  const unsigned idx = ki + ko * conv_N1 + (ni-P) * S + (no-P) * S * conv_N1 + i*conv_N1*conv_N1;
191  const unsigned idx1 = j*F*F+ko*F+ki;
192  const unsigned idx2 = no*conv_N2+ni+i*conv_N2*conv_N2;
193  conv_matrix(idx1,idx2) = conv_layer(j,idx);
194  }
195  }
196  }
197  }
198  }

300 void conv2Layer(const unsigned n_img,
301  const unsigned N,
302  const MyMatrix &conv_obj,
303  MyMatrix &layer){
304 
305  const unsigned n_filter = conv_obj.rows();
306  const unsigned filter_size = conv_obj.cols()/n_img;
307 
308  layer.resize(n_filter*filter_size,n_img);
309 
310  for(unsigned i = 0; i < n_img; i++){
311  MyMatrix sub_conv = conv_obj.block(0, i*N*N, n_filter, N*N).transpose();
312  layer.col(i) = Map<MyVector>(sub_conv.data(), sub_conv.cols()*sub_conv.rows());
313  }
314  layer.transposeInPlace();
315 }
