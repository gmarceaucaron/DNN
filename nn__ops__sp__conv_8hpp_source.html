109  const unsigned n_act0,
110  const unsigned n_act1,
111  const double sigma,
112  unsigned &param_counter,
113  std::vector<SpEntry> &coeff){
114 
115  assert(sparsity < n_act0);
116 
117  std::uniform_real_distribution<double> uniform(0.0,1.0);
118 
119  // Generate randomly the sparse network
120  for(unsigned i = 0; i < n_act1; i++){
121  double u = 0.;
122  unsigned t = 0; // total input records dealt with
123  unsigned m = 0; // number of items selected so far
124 
125  // Knuth's Algorithm of sampling without replacement

127  u = uniform(gen); // call a uniform(0,1) random number generator

129  t++;
130  }

206 int initNetwork(const std::vector<unsigned> &nn_arch,
207  const std::string act_func,
208  const unsigned sparsity,
209  const std::vector<ConvLayerParams> &conv_params,
210  const std::vector<PoolLayerParams> &pool_params,
211  MyMatrix &W_out,
212  std::vector<MySpMatrix> &W,
213  std::vector<MySpMatrix> &Wt,
214  std::vector<MyVector> &B,
215  std::vector<MyMatrix> &convW,
216  std::vector<MyMatrix> &convW_T,
217  std::vector<MyVector> &convB){
218 
219  // Fix some parameters
220  const unsigned n_layers = nn_arch.size();
221  double sigma = 1.0;
222  if(act_func=="sigmoid"){
223  sigma = 4.0;
224  }
225  unsigned param_counter = 0;
226 
227  // Initialize the convolutional layer
228  param_counter += initConvLayer(conv_params, convW, convW_T, convB);
229 
230  // Initialize the sparse weights
231  for(unsigned i = 0; i < n_layers-2; i++){
232 
233  std::vector<SpEntry> coeff(nn_arch[i+1]*sparsity);
234  initSparseLayer(sparsity, nn_arch[i], nn_arch[i+1], sigma, param_counter, coeff);
235  param_counter += nn_arch[i+1];
236 
237  MySpMatrix spW(nn_arch[i], nn_arch[i+1]);
238  spW.setFromTriplets(coeff.begin(), coeff.end());
239  W[i] = spW;
240  MySpMatrix spWt = spW.transpose();
241  Wt[i] = spWt;
242 
243  if(act_func=="sigmoid"){
244  MyMatrix temp = spW;
245  MyVector b = -0.5 * temp.colwise().sum();
246  B[i] = b;
247  }else if(act_func=="tanh"){
248  MyVector b = MyVector::Zero(W[i].cols());
249  B[i] = b;
250  }else if(act_func=="relu"){ // TODO: find the right way to initialize relu
251  MyVector b = MyVector::Zero(W[i].cols());
252  B[i] = b;
253  }else{
254  std::cout << "Not implemented!" << std::endl;
255  assert(false);
256  }
257  }
258 
259  // Initialize the output weights
260  initFullLayer(nn_arch[n_layers-2], nn_arch[n_layers-1], sigma, W_out);
261  if(act_func=="sigmoid"){
262  B[n_layers-2] = -0.5 * W_out.colwise().sum();
263  }else if(act_func=="tanh"){
264  B[n_layers-2] = MyVector::Zero(W_out.cols());
265  }else if(act_func=="relu"){ // TODO: find the right way to initialize relu
266  B[n_layers-2] = MyVector::Zero(W_out.cols());
267  }else{
268  std::cout << "Not implemented!" << std::endl;
269  assert(false);
270  }
271 
272  param_counter += nn_arch[n_layers-2] * nn_arch[n_layers-1] + nn_arch[n_layers-1];
273  return param_counter;
274

338  MyMatrix& out){
339 
340  MyVector max_coeff = a.rowwise().maxCoeff();
341  MyMatrix centered_a = a.colwise() - max_coeff;
342 
343  // Compute the exponential of the entries
344  for(unsigned j = 0; j < a.cols(); j++){ // Optimization for column-major
345  for(unsigned i = 0; i < a.rows(); i++){
346  centered_a(i,j) = exp(centered_a(i,j));
347  }
348  }
349  MyVector sum_coeff = centered_a.rowwise().sum();
350  out = centered_a.array().colwise() / sum_coeff.array();
