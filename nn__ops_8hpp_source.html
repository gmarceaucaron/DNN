95 int initNetwork(const std::vector<unsigned> &nn_arch,
96  const std::string act_func,
97  std::vector<MyMatrix> &W,
98  std::vector<MyVector> &B){
99 
100  // Determine the noise amplitude for the initialization of the weight matrix
101  const unsigned n_layers = nn_arch.size();
102  double sigma = 1.0;
103  if(act_func=="sigmoid"){
104  sigma = 4.0;
105  }
106  unsigned param_counter = 0;
107 
108  // Initialize the weights according to the chosen activation function
109  for(unsigned i = 0; i < n_layers-1; i++){
110  initWeight(nn_arch[i], nn_arch[i+1], sigma, W[i]);
111  if(act_func=="sigmoid"){
112  B[i] = -0.5 * W[i].colwise().sum();
113  }else if(act_func=="tanh"){
114  B[i] = MyVector::Zero(W[i].cols());
115  }else if(act_func=="relu"){ //@TODO: find the optimal way to initialize relu
116  B[i] = MyVector::Zero(W[i].cols());
117  }else{
118  std::cout << "Not implemented!" << std::endl;
119  assert(false);
120  }
121  param_counter += nn_arch[i] * nn_arch[i+1] + nn_arch[i+1];
122  }
123  return param_counter;

180 void softmax(const MyMatrix &a,
181  MyMatrix& out){
182 
183  // Translate the pre-activation for numerical stability
184  MyVector max_coeff = a.rowwise().maxCoeff();
185  MyMatrix centered_a = a.colwise() - max_coeff;
186 
187  // Compute the exponential of the entries
188  for(unsigned j = 0; j < a.cols(); j++){ // Optimization for column-major
189  for(unsigned i = 0; i < a.rows(); i++){
190  centered_a(i,j) = exp(centered_a(i,j));
191  }
192  }
193 
194  // Normalize
195  MyVector sum_coeff = centered_a.rowwise().sum();
196  out = centered_a.array().colwise() / sum_coeff.array();
197 }

301 void dropout(const double prob,
302  MyMatrix &A,
303  MyMatrix &B){
304 
305  // The mask is a realization of (n_example X n_unit) realizations of i.i.d bernoulli trials
306  std::bernoulli_distribution dist(prob);
307  MyMatrix mask(A.rows(),A.cols());
308  for(unsigned j = 0; j < A.cols(); j++){
309  for(unsigned i = 0; i < A.rows(); i++){
310  mask(i,j) = dist(gen);
311  }
312  }
313 
314  // Multiply element-wise
315  A = A.cwiseProduct(mask);
316  B = B.cwiseProduct(mask);
317 }

358 void fprop(const bool dropout_flag,
359  const double dropout_prob,
360  const ActivationFunction &act_func,
361  const std::vector<MyMatrix> &W,
362  const std::vector<MyVector> &B,
363  const MyMatrix &X_batch,
364  std::vector<MyMatrix> &Z,
365  std::vector<MyMatrix> &A,
366  std::vector<MyMatrix> &dA){
367 
368  // Verify if the dropout_flag is activated, if so, then we need
369  // to copy the X_batch in order to do dropout on it
370  const MyMatrix* Atilde = &X_batch;
371  if(dropout_flag){
372  Atilde = new MyMatrix(X_batch);
373  dropout(0.8, *const_cast<MyMatrix *>(Atilde));
374  }
375 
376  // For each layer, forward propagate the data and compute the activations (also perform dropout if flag is set)
377  for(unsigned i = 0; i < A.size(); i++){
378  Z[i] = (*Atilde * W[i]).rowwise() + B[i].transpose();
379  A[i].resize(Z[i].rows(), Z[i].cols());
380  dA[i].resize(Z[i].rows(), Z[i].cols());
381  act_func(Z[i],A[i],dA[i]);
382  if(dropout_flag){
383  dropout(dropout_prob, A[i], dA[i]);
384  if(i==0)
385  delete Atilde;
386  }
387  Atilde = &A[i];
388  }
389 
390  // For the output layer, there is no activation (softmax is done later)
391  Z[A.size()] = (A[A.size()-1] * W[A.size()]).rowwise() + B[A.size()].transpose();
392 
393 }

455 void update(const double eta,
456  const std::vector<MyMatrix> &gradB,
457  const std::vector<MyMatrix> &A,
458  const MyMatrix &X_batch,
459  const std::string regularizer,
460  const double lambda,
461  std::vector<MyMatrix> &W,
462  std::vector<MyVector> &B){
463 
464  const unsigned batch_size = X_batch.rows();
465 
466  // Update the hidden layers
467  for(unsigned i = 0; i < W.size(); i++){
468 
469  const MyMatrix* Atilde = &X_batch;
470  if(i > 0){
471  Atilde = &A[i-1];
472  }
473 
474  // Compute the update of the weights
475  MyMatrix dw = (Atilde->transpose() * gradB[i])/batch_size;
476  updateParam(eta, regularizer, lambda, dw, W[i]);
477 
478  // Compute the update of the bias
479  const MyVector gradB_avg = gradB[i].colwise().sum()/batch_size;
480  updateParam(eta, regularizer, lambda, gradB_avg, B[i]);
481  }

501 void adagradUpdate(const double eta,
502  const std::vector<MyMatrix> &gradB,
503  const std::vector<MyMatrix> &A,
504  const MyMatrix &X_batch,
505  const std::string regularizer,
506  const double lambda,
507  const double mat_reg,
508  const double autocorr,
509  std::vector<MyMatrix> &W,
510  std::vector<MyVector> &B,
511  std::vector<MyMatrix> &mu_dW,
512  std::vector<MyVector> &mu_dB){
513 
514  const unsigned batch_size = X_batch.rows();
515 
516  bool ada_first = true;
517  if(mu_dW[0].size()!=0)
518  ada_first = false;
519 
520  // Update the hidden layers
521  for(unsigned i = 0; i < W.size(); i++){
522 
523  const MyMatrix* Atilde = &X_batch;
524  if(i > 0){
525  Atilde = &A[i-1];
526  }
527 
528  // Compute the update of the weights
529  const MyMatrix dw = (Atilde->transpose() * gradB[i])/batch_size;
530 
531  // Initialize the rolling average or update it
532  if(ada_first || autocorr<1e-6){
533  mu_dW[i] = dw.array().square();
534  }else{
535  mu_dW[i] = autocorr * mu_dW[i].array() + (1.0-autocorr) * dw.array().square();
536  }
537 
538  // Compute the adagrad update rule
539  MyMatrix temp = mu_dW[i].cwiseSqrt().array() + mat_reg;
540  MyMatrix ada_dw = dw.cwiseQuotient(temp);
541 
542  // Update the weight parameters
543  updateParam(eta, regularizer, lambda, ada_dw, W[i]);
544 
545  // Compute the update of the weights
546  const MyVector gradB_avg = gradB[i].colwise().sum() / batch_size;
547 
548  // Initialize the rolling average or update it
549  if(ada_first || autocorr<1e-6){
550  mu_dB[i] = gradB_avg.array().square();
551  }else{
552  mu_dB[i] = autocorr * mu_dB[i].array() + (1.0-autocorr) * gradB_avg.array().square();
553  }
554 
555  // Compute the adagrad update rule
556  MyVector temp2 = mu_dB[i].cwiseSqrt().array() + mat_reg;
557  MyVector ada_dB = gradB_avg.cwiseQuotient(temp2);
558 
559  // Update the bias parameters
560  updateParam(eta, regularizer, lambda, ada_dB, B[i]);
561  }
562 }

575 void computeLoss(const ActivationFunction &act_func,
576  const MyMatrix &X,
577  const MyVector &Y,
578  const std::vector<MyMatrix> &W,
579  const std::vector<MyVector> &B,
580  double &loss,
581  double &accuracy){
582 
583  // Perform the forward propagation algorithm without stocking the intermediate values
584  MyMatrix z = (X * W[0]).rowwise() + B[0].transpose();
585  MyMatrix a(z.rows(), z.cols());
586  MyMatrix ap(z.rows(), z.cols());
587  act_func(z, a, ap);
588 
589  for(unsigned i = 1; i < W.size()-1; i++){
590  z = (a * W[i]).rowwise() + B[i].transpose();
591  a.resize(z.rows(), z.cols());
592  ap.resize(z.rows(), z.cols());
593 
594  act_func(z, a, ap); // TODO: replace without the computation of the derivative
595  }
596 
597  a = (a*W[W.size()-1]).rowwise() + B[W.size()-1].transpose();
598 
599  // Compute the output of the network
600  MyMatrix out;
601  softmax(a, out);
602 
603  // Compute the cumulative negative log-likelihood
604  double cum_loss = 0.;
605  for(unsigned i = 0; i < out.rows(); i++){
606  if(out(i,Y(i)) > 1e-6){
607  cum_loss += -1. * log(out(i,Y(i)));
608  }else{
609  cum_loss += 15.0;
610  }
611  }
612 
613  // Compute the accuracy
614  unsigned n_correct = 0;
615  for(unsigned i = 0; i < out.rows(); i++){
616  MyMatrix::Index idx;
617  double val = out.row(i).maxCoeff(&idx);
618  if(idx==Y(i))
619  n_correct++;
620  }
621 
622  // Compute the averages
623  accuracy = (float)n_correct/out.rows();
624  loss = cum_loss/out.rows();
625 }
